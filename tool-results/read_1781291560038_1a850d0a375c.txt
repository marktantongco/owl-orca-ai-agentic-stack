    68→/** Masks sensitive key values for display: 'orca-racer' becomes 'or•••••cer' */
    69→function maskKey(key: string): string {
    70→  if (key === '(internal)') return key
    71→  if (key.length <= 4) return key.charAt(0) + '••' + key.charAt(key.length - 1)
    72→  return key.slice(0, 2) + '•'.repeat(key.length - 4) + key.slice(-2)
    73→}

    75→/** Safely checks clipboard API availability */
    76→function isClipboardAvailable(): boolean {
    77→  return typeof navigator !== 'undefined' && typeof navigator.clipboard !== 'undefined' && typeof navigator.clipboard.writeText === 'function'
    78→}
    79→
    80→/** Sanitize search input — strip potential XSS vectors */
    81→function sanitizeSearch(input: string): string {
    82→  return input.replace(/[<>"'&]/g, '').slice(0, 200)
    83→}

    87→function CopyButton({ text, label }: { text: string; label?: string }) {
    88→  const [copied, setCopied] = useState(false)
    89→  const handleCopy = () => {
    90→    if (!isClipboardAvailable()) {
    91→      toast.error('Clipboard not available in this environment')
    92→      return
    93→    }
    94→    navigator.clipboard.writeText(text).then(() => {
    95→      setCopied(true)
    96→      toast.success(label ? `${label} copied!` : 'Copied to clipboard!')
    97→      setTimeout(() => setCopied(false), 2000)
    98→    }).catch(() => {
    99→      toast.error('Failed to copy')
   100→    })
   101→  }
   102→  return (
   103→    <motion.button
   104→      onClick={handleCopy}
   105→      whileHover={{ scale: 1.05 }}
   106→      whileTap={{ scale: 0.97 }}
   107→      aria-label={copied ? 'Copied' : `Copy: ${text.slice(0, 30)}...`}
   108→      className="inline-flex items-center gap-1 px-2.5 py-1.5 text-xs rounded-lg bg-amber-100 text-amber-800 hover:bg-amber-200 transition-colors dark:bg-amber-900/30 dark:text-amber-400 dark:hover:bg-amber-900/50 min-h-[44px] focus:ring-2 focus:ring-amber-500 focus:ring-offset-2 cursor-pointer"
   109→    >
   110→      {copied ? <Check className="w-3 h-3" /> : <Copy className="w-3 h-3" />}
   111→      {copied ? 'Copied!' : 'Copy'}
   112→    </motion.button>
   113→  )
   114→}

   116→/** Key reveal button for masked sensitive values */
   117→function KeyReveal({ value }: { value: string }) {
   118→  const [revealed, setRevealed] = useState(false)
   119→  if (value === '(internal)') return <span className="text-gray-700 dark:text-gray-300">{value}</span>
   120→  return (
   121→    <motion.button
   122→      onClick={() => setRevealed(!revealed)}
   123→      whileTap={{ scale: 0.97 }}
   124→      className="font-mono cursor-pointer hover:underline focus:ring-2 focus:ring-amber-500 focus:ring-offset-2 rounded"
   125→      aria-label={revealed ? 'Hide API key' : 'Reveal API key'}
   126→      title={revealed ? 'Click to hide' : 'Click to reveal'}
   127→    >
   128→      <span className={revealed ? 'text-gray-700 dark:text-gray-300' : 'text-gray-500 dark:text-gray-400'}>
   129→        {revealed ? value : maskKey(value)}
   130→      </span>
   131→    </motion.button>
   132→  )
