    36→function CopyButton({ text }: { text: string }) {
    37→  const [copied, setCopied] = useState(false)
    38→  const handleCopy = () => {
    39→    navigator.clipboard.writeText(text)
    40→    setCopied(true)
    41→    setTimeout(() => setCopied(false), 2000)
    42→  }
    43→  return (
    44→    <button
    45→      onClick={handleCopy}
    46→      className="inline-flex items-center gap-1 px-2 py-1 text-xs rounded-md bg-amber-100 text-amber-800 hover:bg-amber-200 transition-colors dark:bg-amber-900/30 dark:text-amber-400 dark:hover:bg-amber-900/50"
    47→    >
    48→      {copied ? <Check className="w-3 h-3" /> : <Copy className="w-3 h-3" />}
    49→      {copied ? 'Copied!' : 'Copy'}
    50→    </button>
    51→  )
    52→}
